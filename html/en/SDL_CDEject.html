NAME
       SDL_CDEject - Ejects a CDROM

SYNOPSIS
       #include "SDL.h"

       int SDL_CDEject(SDL_CD *cdrom);

DESCRIPTION
       Ejects the given cdrom.

RETURN VALUE
       Returns 0 on success, or -1 on an error.

       SDL_CD
