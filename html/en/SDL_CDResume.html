NAME
       SDL_CDResume - Resumes a CDROM

SYNOPSIS
       #include "SDL.h"

       int SDL_CDResume(SDL_CD *cdrom);

DESCRIPTION
       Resumes play on the given cdrom.

RETURN VALUE
       Returns 0 on success, or -1 on an error.

SEE ALSO
       SDL_CDPlay, SDL_CDPause
