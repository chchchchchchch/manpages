NAME
       aa_recommendlowkbd  - Add the given driver to the end of list of keyboard
       recommended drivers.

SYNOPSIS
       #include <aalib.h>

       void aa_recommendlowkbd(const char *name);

PARAMETERS
       const char *name
              Name of the driver (ought to match the "shortname"  field  of  the
              driver definition structure).

DESCRIPTION
       Add the given driver to the end of list of keyboard recommended drivers.
